let currentProject = projects[0];
let currentIndex = 0;
let autoSlideInterval = null;
let autoSlidePausedTimeout = null;

function renderSlides() {
    const stage = document.getElementById("slider-stage");
    const dots = document.getElementById("slider-dots");

    stage.innerHTML = currentProject.images.map((src, index) => `
        <div class="slide ${index === currentIndex ? 'active' : ''}">
            <img src="${src}" alt="${currentProject.title}">
        </div>
    `).join("");

    dots.innerHTML = currentProject.images.map((_, index) => `
        <span class="slider-dot ${index === currentIndex ? 'active' : ''}" onclick="goToImage(${index})"></span>
    `).join("");
}

function renderProject() {
    document.getElementById("detail-title").innerText = currentProject.title;
    document.getElementById("detail-text").innerText = currentProject.text;
    document.getElementById("detail-city").innerText = currentProject.city;
    document.getElementById("detail-work").innerText = currentProject.work;

    document.getElementById("detail-objects").innerHTML =
        currentProject.objects.map(item => `<span class="obj-item">${item}</span>`).join("");

    renderSlides();
}

function openProject(project) {
    currentProject = project;
    currentIndex = 0;
    renderProject();
    restartAutoSlide();

    const detailBlock = document.getElementById("project-detail");
    const header = document.getElementById("site-header");
    const headerHeight = header ? header.offsetHeight : 0;

    const targetY =
        detailBlock.getBoundingClientRect().top +
        window.pageYOffset -
        headerHeight -
        10;

    window.scrollTo({
        top: targetY,
        behavior: "smooth"
    });
}

function nextImage() {
    currentIndex = (currentIndex + 1) % currentProject.images.length;
    renderSlides();
    pauseAutoSlideTemporarily();
}

function prevImage() {
    currentIndex = (currentIndex - 1 + currentProject.images.length) % currentProject.images.length;
    renderSlides();
    pauseAutoSlideTemporarily();
}

function goToImage(index) {
    currentIndex = index;
    renderSlides();
    pauseAutoSlideTemporarily();
}

function startAutoSlide() {
    stopAutoSlide();
    autoSlideInterval = setInterval(() => {
        currentIndex = (currentIndex + 1) % currentProject.images.length;
        renderSlides();
    }, 3500);
}

function stopAutoSlide() {
    if (autoSlideInterval) {
        clearInterval(autoSlideInterval);
        autoSlideInterval = null;
    }
}

function restartAutoSlide() {
    if (autoSlidePausedTimeout) {
        clearTimeout(autoSlidePausedTimeout);
        autoSlidePausedTimeout = null;
    }
    startAutoSlide();
}

function pauseAutoSlideTemporarily() {
    stopAutoSlide();

    if (autoSlidePausedTimeout) {
        clearTimeout(autoSlidePausedTimeout);
    }

    autoSlidePausedTimeout = setTimeout(() => {
        startAutoSlide();
    }, 6000);
}

document.addEventListener("DOMContentLoaded", function() {
    renderProject();
    startAutoSlide();
});
